		function miConcat(){
        		
        	var vecUno = document.getElementById("vectorUno");

       		var vecUnoValor = vecUno.value;
 	  		var vecDosValor = vecDos.value;

       		var vecDosArray = vecDosValor.split("");

        	var longVecUno = 0;
        	var longVecDos = 0;
        		
        	for (var i in vecUnoArray){
        		longVecUno++;
        	}
        		
        	for (var i in vecDosArray){
        		longVecDos++;
        	}
        		        		
        	var longAux = longVecUno + longVecDos;
        		 
        	var vecTres = new Array(longAux);
        		
    		for (var i = 0; i < longVecUno; i++){
    			vecTres[i] = vecUnoArray[i];
    		}
        		
        	var aux = 0;
        	for (var index = longVecUno ; index < longAux; index++ ){
        		vecTres[index] = vecDosArray[aux];
        		aux++;
        	}
        	
        	alert(vecTres.toString());
        }

		function miReverse(){
			
			var mensaje = document.getElementById("cadena");
 	  		var mensajeValor = mensaje.value;
        	var longitud = 0;
        	var mensajeArray = mensajeValor.split("");
     		for(var i in mensajeArray){
     			longitud++;
     		}
     		var textoInvertido = "";
     		var resultado = new Array(longitud);
     		var a = longitud-1;
     		for(var i = 0; i < longitud; i++){
     			resultado[a] = mensajeArray[i];
     			a--;
     			   			   			
     		}
     		
     		for (var index in resultado) {
				textoInvertido = textoInvertido + resultado[index];
			}		
			alert("El reverse de la cadena - " + mensajeValor + " - es: " + textoInvertido)
     	}

	function miSplit(cadena, separador){	

 		cadena=document.getElementById("caden");
		cadena=cadena.value;
		separador=document.getElementById("separador");
		separador=separador.value;
		var longitud=0;
		var vector=new Array();
		var posicion=0;
		
		for (var i in cadena){
			longitud++;	
		}
			
		vector[posicion]="";
		for(var i = 0; i < longitud; i++){
			if(cadena[i]!=separador){
				vector[posicion]+=cadena[i];
			}
			else{
				posicion++;
				vector[posicion]="";
			}	
		}
		alert (vector.toString());

	}
